524 inline RCP<const Set> imageset(const RCP<const Basic> &sym,
525  const RCP<const Basic> &expr,
526  const RCP<const Set> &base)
527 {
528  if (not is_a_sub<Symbol>(*sym))
529  throw SymEngineException("first arg is expected to be a symbol");
530 
531  if (eq(*expr, *sym) or eq(*base, *emptyset()))
532  return base;
533 
534  if (is_a_Number(*expr))
535  return finiteset({expr});
536  if (is_a_Set(*expr)) {
537  for (const auto &s : static_cast<const Set &>(*expr).get_args()) {
538  if (not(is_a_Number(*s) or is_a<Constant>(*s)
539  or is_a_Boolean(*s))) {
540  return make_rcp<const ImageSet>(sym, expr, base);
541  }
542  }
543  return finiteset({expr});
544  }
545 
546  if (is_a<FiniteSet>(*base)) {
547  map_basic_basic d;
548  set_basic temp;
549  for (const auto &s :
550  down_cast<const FiniteSet &>(*base).get_container()) {
551  d[sym] = s;
552  temp.insert(expr->subs(d));
553  d.clear();
554  }
555  return finiteset(temp);
556  }
557 
558  if (is_a<ImageSet>(*base)) {
559  const ImageSet &imbase = down_cast<const ImageSet &>(*base);
560  map_basic_basic d;
561  d[sym] = imbase.get_expr();
562  return imageset(imbase.get_symbol(), expand(expr->subs(d)),
563  imbase.get_baseset());
564  }
565 
566  return make_rcp<const ImageSet>(sym, expr, base);
567 }
