31int Basic::__cmp__(const Basic &o) const
32{
33 auto a = this->get_type_code();
34 auto b = o.get_type_code();
35 if (a == b) {
36 return this->compare(o);
37 } else {
38 // We return the order given by the numerical value of the TypeID enum

41 // to be ordered.
