140 void bvisit(const Add &x)
141 {
142 umap_basic_num dict;
143 RCP<const Number> coef = zero;
144 for (auto &p : x.get_dict()) {
145 p.first->accept(*this);
146 if (neq(*coeff_, *zero)) {
147 Add::coef_dict_add_term(outArg(coef), dict, p.second, coeff_);
148 }
149 }
150 if (eq(*zero, *n_)) {
151 iaddnum(outArg(coef), x.get_coef());
152 }
153 coeff_ = Add::from_dict(coef, std::move(dict));
154 }
155
156 void bvisit(const Mul &x)
157 {
158 for (auto &p : x.get_dict()) {
159 if (eq(*p.first, *x_) and eq(*p.second, *n_)) {
160 map_basic_basic dict = x.get_dict();
161 dict.erase(p.first);
162 coeff_ = Mul::from_dict(x.get_coef(), std::move(dict));
163 return;
164 }
165 }
166 if (eq(*zero, *n_) and not has_symbol(x, *x_)) {
167 coeff_ = x.rcp_from_this();
168 } else {
169 coeff_ = zero;
170 }
171 }
172
173 void bvisit(const Pow &x)
174 {
175 if (eq(*x.get_base(), *x_) and eq(*x.get_exp(), *n_)) {
176 coeff_ = one;
177 } else if (neq(*x.get_base(), *x_) and eq(*zero, *n_)) {
178 coeff_ = x.rcp_from_this();
179 } else {
180 coeff_ = zero;
181 }
182 }
