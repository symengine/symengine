26 void mp_fdiv_qr(integer_class &q, integer_class &r, const integer_class &a,
27  const integer_class &b)
28 {
29  /*boost::multiprecision doesn't have a built-in fdiv_qr (floored division).
30  Its divide_qr uses truncated division, as does its
31  modulus operator. Thus, using boost::multiprecision::divide_qr we get:
32  divide_qr(-5, 3, quo, rem) //quo == -1, rem == -2
33  divide_qr(5, -3, quo, rem) //quo == -1, rem == 2
34  but we want:
35  mp_fdiv_r(quo, rem, -5, 3) //quo == -2, rem == 1
36  mp_fdiv_r(quo, rem, 5, -3) //rem == -2, rem == -1
37  The problem arises only when the quotient is negative. To convert
38  a truncated result into a floored result in this case, simply subtract
39  one from the truncated quotient and add the divisor to the truncated
40  remainder.
41  */
42 
43  // must copy a and b before calling divide_qr because a or b may refer to
44  // the same
45  // object as q or r, causing incorrect results
46  integer_class a_cpy = a, b_cpy = b;
47  bool neg_quotient = ((a < 0 && b > 0) || (a > 0 && b < 0)) ? true : false;
48  boost::multiprecision::divide_qr(a_cpy, b_cpy, q, r);
49  // floor the quotient if necessary
50  if (neg_quotient && r != 0) {
51  q -= 1;
52  }
53  // remainder should have same sign as divisor
54  if ((b_cpy > 0 && r < 0) || (b_cpy < 0 && r > 0)) {
55  r += b_cpy;
56  return;
57  }
58 }

77 void mp_gcdext(integer_class &gcd, integer_class &s, integer_class &t,
78  const integer_class &a, const integer_class &b)
79 {
80 
81  integer_class this_s(1);
82  integer_class this_t(0);
83  integer_class next_s(0);
84  integer_class next_t(1);
85  integer_class this_r(a);
86  integer_class next_r(b);
87  integer_class q;
88  while (next_r != 0) {
89  // should use truncated division, so use
90  // boost::multiprecision::divide_qr
91  // beware of overwriting this_r during internal operations of divide_qr
92  // copy it first
93  integer_class this_r_cpy = this_r;
94  boost::multiprecision::divide_qr(this_r_cpy, next_r, q, this_r);
95  this_s -= q * next_s;
96  this_t -= q * next_t;
97  std::swap(this_s, next_s);
98  std::swap(this_t, next_t);
99  std::swap(this_r, next_r);
100  }
101  // normalize the gcd, s and t
102  if (this_r < 0) {
103  this_r *= -1;
104  this_s *= -1;
105  this_t *= -1;
106  }
107  gcd = std::move(this_r);
108  s = std::move(this_s);
109  t = std::move(this_t);
110 }

150 void mp_powm(integer_class &res, const integer_class &base,
151  const integer_class &exp, const integer_class &m)
152 {
153  // if exp is negative, interpret as follows
154  // base**(exp) mod m == (base**(-1))**abs(exp) mod m
155  // == (base**(-1) mod m) ** abs(exp) mod m
156  // where base**(-1) mod m is the modular inverse
157  if (exp < 0) {
158  integer_class base_inverse;
159  if (!mp_invert(base_inverse, base, m)) {
160  throw SymEngine::SymEngineException("negative exponent undefined "
161  "in powm if base is not "
162  "invertible mod m");
163  }
164  res = boost::multiprecision::powm(base_inverse, mp_abs(exp), m);
165  return;
166  } else {
167  res = boost::multiprecision::powm(base, exp, m);
168  // boost's powm calculates base**exp % m, but uses truncated
169  // modulus, e.g. powm(-2,3,5) == -3. We want powm(-2,3,5) == 2
170  if (res < 0) {
171  res += m;
172  }
173  }

411 bool mp_perfect_power_p(const integer_class &i)
412 {
413  if (i == 0 || i == 1 || i == -1) {
414  return true;
415  }
416  // if i == a**k, with k == pq for some integers p,q
417  // then i == a**(pq) == (a**p)**q == (a**q)**p
418  // hence i is a pth power and a qth power
419  // Hence if i == p**k, then i is a pth power for any p
420  // in the prime factorization of k, and it suffices
421  // to check whether i is a prime power.

423  // the largest possible prime p would arise from the
424  // case where i is a prime power of two
425  // so check all prime roots up to log(i) base 2.

430  // with an even root of a negative

432  return true;
433  }

435  integer_class root(0);
436  while (true) {
437  mp_nextprime(p, p);
438  if (p > max) {

440  }
441  if (mp_root(root, i, p.convert_to<unsigned long>())) {

479 // private function that computes jacobi symbols
480 // without checking that arguments satisfy a >= 0
481 // and n is odd.
482 int unchecked_jacobi(const integer_class &a, const integer_class &n)
483 {
484  // https://en.wikipedia.org/wiki/Jacobi_symbol#Calculating_the_Jacobi_symbol
485  if (a == 1) {
486  return 1;
487  }
488  integer_class num = a;
489  integer_class den = n;
490  // (1) Reduce the "numerator" modulo the "denominator"
491  num = fmod(num, den);
492 
493  // (2) Extract any factors of 2 from the "numerator"
494  unsigned long factors_of_two = 0;
495  while (num % 2 == 0 && num != 0) {
496  num /= 2; // use a shift instead of division here? faster?
497  ++factors_of_two;
498  }
499  int product_of_twos = 1; // (2 | den)**factors_of_two
500 
501  // (2 | den) is -1 iff den % 8 = 3 or den % 8 = 5.
502  // If factors_of_two is odd and (2 | den) is -1,
503  // then (2 | den)**factors_of_two is -1.
504  // Otherwise, (2 | den)**factors_of_two is 1.
505  int den_mod_8 = fmod(den, 8).convert_to<int>();
506  if ((factors_of_two % 2 == 1) && (den_mod_8 == 3 || den_mod_8 == 5)) {
507  product_of_twos = -1;
508  }
509 
510  // (3) If the remaining "numerator" (after extraction of twos) is 1,
511  // then the remaining jacobi symbol is 1.
512  // If the "numerator" and "denominator" are not coprime, result is 0.
513  if (num == 1) {
514  return product_of_twos;
515  }
516  if (boost::multiprecision::gcd(num, den) != 1) {
517  return 0;
518  }
519 
520  // (4) Otherwise, the "numerator" and "denominator" are now odd
521  // positive coprime integers, so we can flip the symbol
522  // with quadratic reciprocity, then return to step (1)
523 
524  // (num | den) == (den | num), unless num % 4 and den %4 are both
525  // 3, in which case (num | den) == (-1) * (den | num)
526  int quadratic_reciprocity_factor = 1;
527  if ((fmod(num, 4) == 3) && (fmod(den, 4) == 3)) {
528  quadratic_reciprocity_factor = -1;
529  }
530  return product_of_twos * quadratic_reciprocity_factor
531  * unchecked_jacobi(den, num);
532 }

546 int mp_kronecker(const integer_class &a, const integer_class &n)
547 {
548  /*
549  https://en.wikipedia.org/wiki/Kronecker_symbol
550  We compute the Kronecker symbol in terms of the Jacobi symbol.
551  For an integer n!=0, let n = u * PRODUCT (p_i**e_i), where u = -1 or 1 and
552  the p_i's and e_i's are the primes and corresponding powers
553  in the prime factorization of |n|.
554  Then for an integer a, the Kronecker symbol is given by
555  (a | n) = (a | u) * PRODUCT [(a | p_i)**e_i]
556  where
557  (a | u) is -1 if u is -1 and a < 0. Otherwise it is 1.
558  (a | 2) is 0 if a is even, 1 if (a mod 8) == 1 or 7, and -1 if (a mod 8) ==
559  3 or 5.
560  (a | p) is the Legendre symbol if p is an odd prime.
561 
562  Let j be the power of the prime 2 in n's prime factorization, and let
563  m = |n|/(2**j) -- that is, m is |n| with all factors of 2 extracted.
564 
565  Notice that, if p_1 is 2, then
566  PRODUCT [(a | p_i)**e_i]
567  == (a | 2)**j * PRODUCT [(a | p_i)**e_i] here the p_i's are the remaining
568  (odd) prime factors
569  == (a | 2)**j * (a | m), here (a | m) is the Jacobi
570  symbol, because m>0 is odd
571 
572  Thus,
573  (a | n) == (a | u) * (a | 2)**j * (a | m) if n is even
574  (a | n) == (a | u) * (a | m) if n is odd
575  */
576 
577  if (n == 0) {
578  throw std::runtime_error("second arg of Kronecker cannot be zero");
579  }
580 
581  // Compute (a | u)
582  int kr_a_u = 1;
583  if (n.sign() == -1 && a < 0) {
584  kr_a_u = -1;
585  }
586 
587  // Compute m, j
588  integer_class m = boost::multiprecision::abs(n);
589  unsigned long j = 0;
590  while (m % 2 == 0 && m != 0) { // while m is even
591  m /= 2; // implement with shift? faster?
592  ++j;
593  }
594 
595  // if n is even, compute (a | 2)**j
596  int kr_a_2 = 0;
597  int kr_a_2_to_j = 0;
598  int a_mod_8 = fmod(a, 8).convert_to<int>();
599 
600  if (a % 2 != 0) {
601  kr_a_2 = (a_mod_8 == 1 || a_mod_8 == 7) ? 1 : -1;
602  kr_a_2_to_j = (kr_a_2 == -1 && (j % 2 != 0))
603  ? -1
604  : 1; //(-1)**odd == -1; (-1)**even=(1)**integer=1
605  }
606 
607  if (n % 2 == 0) {
608  return kr_a_u * kr_a_2_to_j * unchecked_jacobi(a, m);
609  } else {
610  return kr_a_u * unchecked_jacobi(a, m);
611  }
612 }
