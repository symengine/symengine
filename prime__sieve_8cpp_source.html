49  unsigned start = _primes.back() + 1;
50  if (limit <= start)
51  return;

54  start = _primes.back() + 1;

57  unsigned segment = _sieve_size;
58  std::valarray<bool> is_prime(segment);
59  for (; start <= limit; start += 2 * segment) {
60  unsigned finish = std::min(start + segment * 2 + 1, limit);
61  is_prime[std::slice(0, segment, 1)] = true;
62  // considering only odd integers. An odd number n corresponds to
63  // n-start/2 in the array.
64  for (unsigned index = 1; index < _primes.size()
65  and _primes[index] * _primes[index] <= finish;
66  ++index) {
67  unsigned n = _primes[index];
68  unsigned multiple = (start / n + 1) * n;
69  if (multiple % 2 == 0)
70  multiple += n;
71  if (multiple > finish)
72  continue;
73  std::slice sl = std::slice((multiple - start) / 2,
74  1 + (finish - multiple) / (2 * n), n);
75  // starting from n*n, all the odd multiples of n are marked not
76  // prime.
77  is_prime[sl] = false;
78  }
79  for (unsigned n = start + 1; n <= finish; n += 2) {
80  if (is_prime[(n - start) / 2])
81  _primes.push_back(n);
82  }
83  }
